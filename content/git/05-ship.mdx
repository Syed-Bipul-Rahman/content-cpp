# Ship mygit

At this point `mygit` can snapshot a directory, commit, branch, and merge.
The last mile is communicating with the outside world: push to GitHub,
fetch from a teammate's machine, and prove it all works with a test suite.

## The smart HTTP protocol

Git has two transports: the legacy dumb HTTP (read-only static files) and
smart HTTP (`git-upload-pack` / `git-receive-pack`). We implement the
smart variant because it's what GitHub, GitLab, and every modern host
expects.

### Discovery

```
GET <url>/info/refs?service=git-upload-pack   (for fetch/clone)
GET <url>/info/refs?service=git-receive-pack  (for push)
```

Response headers include `Content-Type: application/x-git-upload-pack-advertisement`.
The body is a sequence of **pkt-line** frames.

### pkt-line

Everything in smart HTTP is wrapped in pkt-line:

```
0032<32 hex digits of data length including this 4-byte header>\n
0000         <- flush packet
```

A `0000` flush marks the end of a group. Implementation:

```cpp
// src/pktline.hpp
#pragma once
#include <string>
#include <vector>

namespace mygit {

// Encode a single pkt-line.
std::string pkt_encode(std::string_view data);
std::string pkt_flush();

// Decode a stream of pkt-lines; returns payloads (strips \n), empty vec on flush.
std::vector<std::string> pkt_decode_all(std::string_view stream);

}
```

```cpp
// src/pktline.cpp
#include "pktline.hpp"
#include <cstdio>
#include <stdexcept>

namespace mygit {

std::string pkt_encode(std::string_view data) {
    char hdr[5];
    std::snprintf(hdr, sizeof(hdr), "%04zx",
                  static_cast<std::size_t>(data.size() + 4));
    return std::string(hdr, 4) + std::string(data);
}

std::string pkt_flush() { return "0000"; }

std::vector<std::string> pkt_decode_all(std::string_view s) {
    std::vector<std::string> out;
    std::size_t i = 0;
    while (i + 4 <= s.size()) {
        char hdr[5] = {}; std::memcpy(hdr, s.data() + i, 4);
        std::size_t len = std::stoul(std::string(hdr, 4), nullptr, 16);
        if (len == 0) break;   // flush
        if (len < 4 || i + len > s.size()) throw std::runtime_error("short pkt");
        std::string payload(s.data() + i + 4, len - 4);
        if (!payload.empty() && payload.back() == '\n') payload.pop_back();
        out.push_back(std::move(payload));
        i += len;
    }
    return out;
}

}
```

### HTTP client with libcurl

```cpp
// src/http.hpp
#pragma once
#include <string>
#include <string_view>
#include <vector>

namespace mygit {

struct HttpResponse {
    int         status{0};
    std::string body;
    std::string content_type;
};

HttpResponse http_get (const std::string& url,
                       const std::vector<std::string>& headers = {});
HttpResponse http_post(const std::string& url,
                       std::string_view   body,
                       const std::string& content_type,
                       const std::vector<std::string>& headers = {});

}
```

```cpp
// src/http.cpp  — thin libcurl wrapper
#include "http.hpp"
#include <curl/curl.h>
#include <stdexcept>

namespace mygit {

static std::size_t write_cb(char* ptr, std::size_t sz, std::size_t n, void* ud) {
    static_cast<std::string*>(ud)->append(ptr, sz * n);
    return sz * n;
}

HttpResponse http_get(const std::string& url, const std::vector<std::string>& hdr) {
    auto* c = curl_easy_init();
    if (!c) throw std::runtime_error("curl init");
    HttpResponse r;
    curl_easy_setopt(c, CURLOPT_URL, url.c_str());
    curl_easy_setopt(c, CURLOPT_WRITEFUNCTION, write_cb);
    curl_easy_setopt(c, CURLOPT_WRITEDATA, &r.body);
    curl_slist* sl = nullptr;
    for (const auto& h : hdr) sl = curl_slist_append(sl, h.c_str());
    if (sl) curl_easy_setopt(c, CURLOPT_HTTPHEADER, sl);
    curl_easy_perform(c);
    long code; curl_easy_getinfo(c, CURLINFO_RESPONSE_CODE, &code); r.status = code;
    curl_slist_free_all(sl);
    curl_easy_cleanup(c);
    return r;
}

HttpResponse http_post(const std::string& url, std::string_view body,
                       const std::string& ct,
                       const std::vector<std::string>& hdr) {
    auto* c = curl_easy_init();
    if (!c) throw std::runtime_error("curl init");
    HttpResponse r;
    curl_easy_setopt(c, CURLOPT_URL, url.c_str());
    curl_easy_setopt(c, CURLOPT_POST, 1L);
    curl_easy_setopt(c, CURLOPT_POSTFIELDS, body.data());
    curl_easy_setopt(c, CURLOPT_POSTFIELDSIZE, body.size());
    curl_easy_setopt(c, CURLOPT_WRITEFUNCTION, write_cb);
    curl_easy_setopt(c, CURLOPT_WRITEDATA, &r.body);
    curl_slist* sl = nullptr;
    sl = curl_slist_append(sl, ("Content-Type: " + ct).c_str());
    for (const auto& h : hdr) sl = curl_slist_append(sl, h.c_str());
    curl_easy_setopt(c, CURLOPT_HTTPHEADER, sl);
    curl_easy_perform(c);
    long code; curl_easy_getinfo(c, CURLINFO_RESPONSE_CODE, &code); r.status = code;
    curl_slist_free_all(sl);
    curl_easy_cleanup(c);
    return r;
}

}
```

### `fetch`

```
1. GET /info/refs?service=git-upload-pack
2. Parse advertised refs (branch -> sha mappings).
3. POST /git-upload-pack with "want <sha>" lines.
4. Receive a packfile — parse it and explode into loose objects.
```

The packfile is the hard part: it's a binary stream of delta-compressed
objects. For the educational version, use `git index-pack` to explode it:

```cpp
int cmd_fetch(std::span<const std::string_view> args) {
    if (args.empty()) { std::fprintf(stderr, "fetch: url required\n"); return 2; }
    auto url = std::string{args[0]};
    auto r   = Repo::discover();

    // 1. Discovery
    auto adv = http_get(url + "/info/refs?service=git-upload-pack",
                        {"Git-Protocol: version=2"});
    auto pkts = pkt_decode_all(adv.body);

    // Collect advertised refs (skip first capability line)
    std::map<std::string, std::string> remote_refs;
    for (std::size_t i = 1; i < pkts.size(); ++i) {
        auto& p = pkts[i];
        auto sp = p.find(' ');
        if (sp == std::string::npos) continue;
        remote_refs[p.substr(sp + 1)] = p.substr(0, sp);
    }

    // 2. Negotiate: want every ref we don't have
    std::string upload_req;
    for (const auto& [ref, sha] : remote_refs) {
        if (!resolve_ref(r, ref))
            upload_req += pkt_encode("want " + sha + "\n");
    }
    upload_req += pkt_flush();
    upload_req += pkt_encode("done\n");

    // 3. Receive packfile
    auto pack = http_post(url + "/git-upload-pack", upload_req,
                          "application/x-git-upload-pack-request");

    // 4. Explode with real git (for simplicity)
    auto pack_path = r.git_dir() / "packed.pack";
    std::ofstream{pack_path, std::ios::binary}.write(
        pack.body.data(), static_cast<std::streamsize>(pack.body.size()));
    auto cmd = "git -C " + r.git_dir().parent_path().string() +
               " index-pack -v " + pack_path.string();
    std::system(cmd.c_str());

    // Update refs
    for (const auto& [ref, sha] : remote_refs)
        update_ref(r, ref, sha);

    std::cout << "Fetched " << remote_refs.size() << " refs\n";
    return 0;
}
```

For a fully self-contained implementation, parse the packfile yourself:
it's `PACK` + version (2) + object count + objects (each prefixed with a
type/size header, some as deltas). Each object is a zlib stream. It's
about 200 more lines — the full format is in `Documentation/technical/pack-format.txt`
in the git source tree.

### `push`

```
1. GET /info/refs?service=git-receive-pack
2. Build the list of updates (branch -> old_sha new_sha)
3. POST /git-receive-pack with ref updates + a packfile of new objects
```

```cpp
int cmd_push(std::span<const std::string_view> args) {
    if (args.size() < 2) {
        std::fprintf(stderr, "push: usage: <url> <branch>\n");
        return 2;
    }
    auto url    = std::string{args[0]};
    auto branch = std::string{args[1]};
    auto r      = Repo::discover();

    auto local_sha = resolve_ref(r, "refs/heads/" + branch);
    if (!local_sha) { std::fprintf(stderr, "push: branch not found\n"); return 2; }

    // Discovery
    auto adv = http_get(url + "/info/refs?service=git-receive-pack");
    auto pkts = pkt_decode_all(adv.body);
    std::string remote_sha(40, '0');
    for (std::size_t i = 1; i < pkts.size(); ++i) {
        auto& p = pkts[i];
        if (p.find("refs/heads/" + branch) != std::string::npos)
            remote_sha = p.substr(0, 40);
    }

    // Build request: ref update + packfile
    std::string body;
    body += pkt_encode(remote_sha + ' ' + *local_sha + " refs/heads/" + branch + '\0' + "report-status\n");
    body += pkt_flush();

    // Collect objects reachable from local_sha but not from remote (simplified: send all)
    auto pack_file = r.git_dir() / "push.pack";
    auto mk_pack = "git -C " + r.git_dir().parent_path().string() +
                   " pack-objects --stdout --thin --revs > " + pack_file.string();
    FILE* pp = popen(("echo " + *local_sha + " | " + mk_pack).c_str(), "r");
    if (pp) pclose(pp);

    std::ifstream pf{pack_file, std::ios::binary};
    std::ostringstream ps; ps << pf.rdbuf();
    body += ps.str();

    auto resp = http_post(url + "/git-receive-pack", body,
                          "application/x-git-receive-pack-request");
    std::cout << "Push " << (resp.status == 200 ? "ok" : "failed")
              << " (" << resp.status << ")\n";
    return resp.status == 200 ? 0 : 2;
}
```

## Test suite

### tests/run.sh

```bash
#!/usr/bin/env bash
set -euo pipefail

GIT=./build/mygit
REAL=${REAL_GIT:-git}

T=$(mktemp -d /tmp/mygit-XXXX)
trap 'rm -rf "$T"' EXIT

ok() { echo "ok: $*"; }
fail() { echo "FAIL: $*"; exit 1; }

# init
cd "$T"
"$GIT" init
[[ -d .git/objects ]] || fail "no objects dir"
[[ -f .git/HEAD    ]] || fail "no HEAD"
ok "init"

# hash-object roundtrip
echo "hello world" > a.txt
SHA=$("$GIT" hash-object a.txt)
RSHA=$("$REAL" hash-object a.txt)
[[ "$SHA" == "$RSHA" ]] || fail "hash mismatch: $SHA vs $RSHA"
ok "hash-object matches real git"

# write and cat-file
"$GIT" hash-object -w a.txt
OUT=$("$GIT" cat-file -p "$SHA")
[[ "$OUT" == "hello world" ]] || fail "cat-file wrong: $OUT"
ok "cat-file -p"

TYPE=$("$GIT" cat-file -t "$SHA")
[[ "$TYPE" == "blob" ]] || fail "cat-file -t: $TYPE"
ok "cat-file -t"

# add + commit
"$GIT" add a.txt
"$GIT" commit -m "first"
HEAD=$("$REAL" -C "$T" rev-parse HEAD 2>/dev/null || "$GIT" log | head -1 | cut -d' ' -f2)
[[ -n "$HEAD" ]] || fail "no commit"
ok "commit"

# status — clean
STATUS=$("$GIT" status)
[[ -z "$STATUS" ]] || fail "status not clean: $STATUS"
ok "status clean"

# status — modified
echo "new" >> a.txt
STATUS=$("$GIT" status)
[[ "$STATUS" == *" M a.txt"* ]] || fail "status not showing modification: $STATUS"
ok "status modified"

# branch + checkout + merge
"$GIT" branch feature
"$GIT" checkout feature
echo "feature line" >> a.txt
"$GIT" add a.txt
"$GIT" commit -m "feature"

"$GIT" checkout main
"$GIT" merge feature

CONTENT=$(cat a.txt)
[[ "$CONTENT" == *"feature line"* ]] || fail "merge didn't apply content"
ok "branch/checkout/merge"

# write-tree is real-git-compatible
TREE=$("$GIT" write-tree)
RTREE=$("$REAL" -C "$T" write-tree 2>/dev/null || true)
if [[ -n "$RTREE" ]]; then
    [[ "$TREE" == "$RTREE" ]] || fail "write-tree mismatch: $TREE vs $RTREE"
    ok "write-tree matches real git"
else
    ok "write-tree (real git skipped)"
fi

echo ""
echo "all pass"
```

```bash
chmod +x tests/run.sh
./tests/run.sh
```

### Unit-test the pkt-line codec

```cpp
// tests/test_pktline.cpp  (Catch2)
#include <catch2/catch.hpp>
#include "pktline.hpp"

TEST_CASE("pkt roundtrip") {
    auto enc = mygit::pkt_encode("hello\n");
    REQUIRE(enc.size() == 4 + 6);
    REQUIRE(enc.substr(0, 4) == "000a");   // 4 + 6 = 10 = 0x000a
    auto dec = mygit::pkt_decode_all(enc);
    REQUIRE(dec.size() == 1);
    REQUIRE(dec[0] == "hello");
}

TEST_CASE("flush") {
    auto s = mygit::pkt_flush();
    REQUIRE(s == "0000");
    auto dec = mygit::pkt_decode_all(s);
    REQUIRE(dec.empty());
}
```
