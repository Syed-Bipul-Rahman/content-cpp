# Extending the shell

Now the fun. Pipelines, redirection, real tokenization with quotes, and
environment variable expansion. By the end of this lesson, `msh` is doing
90% of what people use a shell for.

## A better lexer with quoting

A real shell handles `'single'`, `"double"`, escapes, and groups whitespace
inside quotes as one token.

### src/lexer.cpp (replace)

```cpp
#include "lexer.hpp"
#include <cctype>
#include <stdexcept>

namespace msh {

std::vector<std::string> tokenize(std::string_view line) {
    std::vector<std::string> tokens;
    std::string cur;
    bool in_token = false;

    enum class Quote { None, Single, Double };
    Quote q = Quote::None;

    auto flush = [&] {
        if (in_token) { tokens.push_back(std::move(cur)); cur.clear(); in_token = false; }
    };

    for (std::size_t i = 0; i < line.size(); ++i) {
        char c = line[i];

        if (q == Quote::Single) {
            if (c == '\'') { q = Quote::None; }
            else { cur.push_back(c); }
            in_token = true;
            continue;
        }
        if (q == Quote::Double) {
            if (c == '"') { q = Quote::None; }
            else if (c == '\\' && i + 1 < line.size()) { cur.push_back(line[++i]); }
            else { cur.push_back(c); }
            in_token = true;
            continue;
        }

        // not in quotes
        if (c == '\'') { q = Quote::Single; in_token = true; continue; }
        if (c == '"')  { q = Quote::Double; in_token = true; continue; }
        if (c == '\\' && i + 1 < line.size()) { cur.push_back(line[++i]); in_token = true; continue; }
        if (c == '|' || c == '<' || c == '>' || c == '&') {
            flush();
            tokens.emplace_back(1, c);
            continue;
        }
        if (std::isspace(static_cast<unsigned char>(c))) { flush(); continue; }
        cur.push_back(c);
        in_token = true;
    }

    if (q != Quote::None) throw std::runtime_error("unterminated quote");
    flush();
    return tokens;
}

}
```

Now `echo "hello   world"` gives one argument `hello   world`.

## Variable expansion

Replace `$VAR` and `${VAR}` with values from the environment.

### src/expand.hpp

```cpp

}
```

### src/expand.cpp

```cpp
#include "expand.hpp"
#include <cstdlib>
#include <cctype>

namespace msh {

std::string expand(const std::string& s) {
    std::string out;
    out.reserve(s.size());
    for (std::size_t i = 0; i < s.size();) {
        if (s[i] != '$') { out.push_back(s[i++]); continue; }
        // $...
        ++i;
        std::string name;
        if (i < s.size() && s[i] == '{') {
            ++i;
            while (i < s.size() && s[i] != '}') name.push_back(s[i++]);
            if (i < s.size()) ++i;   // skip '}'
        } else {
            while (i < s.size() && (std::isalnum(static_cast<unsigned char>(s[i])) || s[i] == '_')) {
                name.push_back(s[i++]);
            }
        }
        if (const char* v = std::getenv(name.c_str())) out += v;
    }
    return out;
}

}
```

Apply it to every token in `parser.cpp`:

```cpp
#include "expand.hpp"

Command parse(std::vector<std::string> tokens) {
    Command c;
    if (!tokens.empty() && tokens.back() == "&") {
        c.background = true;
        tokens.pop_back();
    }
    c.argv.reserve(tokens.size());
    for (auto& t : tokens) c.argv.push_back(expand(t));
    return c;
}
```

Now `echo $HOME` works.

## Pipelines and redirection — the data model

Bump the `Command` to support all of it:

### src/parser.hpp

```cpp
#pragma once
#include <string>
#include <vector>

namespace msh {

struct SimpleCommand {
    std::vector<std::string> argv;
    std::string in_file;        // < file
    std::string out_file;       // > file
    bool        out_append = false;  // >> file
};

struct Pipeline {
    std::vector<SimpleCommand> stages;
    bool background = false;
};

Pipeline parse(std::vector<std::string> tokens);

}
```

### src/parser.cpp (replace)

```cpp
#include "parser.hpp"
#include "expand.hpp"
#include <stdexcept>

namespace msh {

namespace {

bool is_op(const std::string& t) {
    return t == "|" || t == "<" || t == ">" || t == ">>" || t == "&";
}

}  // anon

Pipeline parse(std::vector<std::string> tokens) {
    Pipeline pipe;

    // collapse > > into >>
    for (std::size_t i = 0; i + 1 < tokens.size(); ++i) {
        if (tokens[i] == ">" && tokens[i + 1] == ">") {
            tokens[i] = ">>";
            tokens.erase(tokens.begin() + i + 1);
        }
    }

    if (!tokens.empty() && tokens.back() == "&") {
        pipe.background = true;
        tokens.pop_back();
    }

    SimpleCommand cur;
    for (std::size_t i = 0; i < tokens.size(); ++i) {
        const auto& t = tokens[i];
        if (t == "|") {
            if (cur.argv.empty()) throw std::runtime_error("syntax: empty pipeline stage");
            pipe.stages.push_back(std::move(cur));
            cur = {};
        } else if (t == "<" || t == ">" || t == ">>") {
            if (i + 1 >= tokens.size()) throw std::runtime_error("syntax: missing redirect target");
            const std::string& target = tokens[++i];
            if      (t == "<")  cur.in_file  = expand(target);
            else if (t == ">")  { cur.out_file = expand(target); cur.out_append = false; }
            else                { cur.out_file = expand(target); cur.out_append = true;  }
        } else if (is_op(t)) {
            throw std::runtime_error("syntax: unexpected operator");
        } else {
            cur.argv.push_back(expand(t));
        }
    }
    if (!cur.argv.empty() || pipe.stages.empty()) {
        if (cur.argv.empty() && pipe.stages.empty()) return pipe;
        pipe.stages.push_back(std::move(cur));
    }
    return pipe;
}

}
```

### src/executor.cpp (replace)

```cpp
#include "executor.hpp"

#include <iostream>
#include <fcntl.h>
#include <sys/wait.h>
#include <unistd.h>
#include <cstdlib>
#include <cstring>
#include <climits>
#include <stdexcept>

namespace msh {

namespace {

bool is_builtin(const std::string& name) {
    return name == "cd" || name == "exit" || name == "pwd" ||
           name == "export" || name == "unset";
}

int run_builtin(const SimpleCommand& cmd) {
    const auto& a = cmd.argv;
    if (a[0] == "exit") std::exit(a.size() > 1 ? std::atoi(a[1].c_str()) : 0);
    if (a[0] == "cd") {
        const char* target = a.size() > 1 ? a[1].c_str() : std::getenv("HOME");
        if (!target) target = "/";
        if (chdir(target) != 0) { std::perror("cd"); return 1; }
        return 0;
    }
    if (a[0] == "pwd") {
        char buf[PATH_MAX];
        if (!getcwd(buf, sizeof(buf))) { std::perror("pwd"); return 1; }
        std::cout << buf << '\n';
        return 0;
    }
    if (a[0] == "export") {
        for (std::size_t i = 1; i < a.size(); ++i) {
            auto eq = a[i].find('=');
            if (eq == std::string::npos) continue;
            setenv(a[i].substr(0, eq).c_str(), a[i].substr(eq + 1).c_str(), 1);
        }
        return 0;
    }
    if (a[0] == "unset") {
        for (std::size_t i = 1; i < a.size(); ++i) unsetenv(a[i].c_str());
        return 0;
    }
    return 1;
}

void apply_redirects(const SimpleCommand& cmd) {
    if (!cmd.in_file.empty()) {
        int fd = open(cmd.in_file.c_str(), O_RDONLY);
        if (fd < 0) { std::perror("open <"); _exit(1); }
        dup2(fd, STDIN_FILENO);
        close(fd);
    }
    if (!cmd.out_file.empty()) {
        int flags = O_WRONLY | O_CREAT | (cmd.out_append ? O_APPEND : O_TRUNC);
        int fd = open(cmd.out_file.c_str(), flags, 0644);
        if (fd < 0) { std::perror("open >"); _exit(1); }
        dup2(fd, STDOUT_FILENO);
        close(fd);
    }
}

std::vector<char*> to_c_argv(const std::vector<std::string>& argv) {
    std::vector<char*> out;
    out.reserve(argv.size() + 1);
    for (const auto& a : argv) out.push_back(const_cast<char*>(a.c_str()));
    out.push_back(nullptr);
    return out;
}

}  // anon

int execute(const Pipeline& pipe) {
    if (pipe.stages.empty()) return 0;

    // single-stage builtin runs in the parent
    if (pipe.stages.size() == 1 && is_builtin(pipe.stages[0].argv[0])
        && pipe.stages[0].in_file.empty() && pipe.stages[0].out_file.empty()
        && !pipe.background) {
        return run_builtin(pipe.stages[0]);
    }

    int prev_read = -1;
    std::vector<pid_t> pids;
    pids.reserve(pipe.stages.size());

    for (std::size_t i = 0; i < pipe.stages.size(); ++i) {
        int pipefd[2] = {-1, -1};
        bool last = (i + 1 == pipe.stages.size());
        if (!last) {
            if (::pipe(pipefd) < 0) { std::perror("pipe"); return 1; }
        }

        pid_t pid = fork();
        if (pid < 0) { std::perror("fork"); return 1; }

        if (pid == 0) {
            // child
            if (prev_read != -1) {
                dup2(prev_read, STDIN_FILENO);
                close(prev_read);
            }
            if (!last) {
                close(pipefd[0]);
                dup2(pipefd[1], STDOUT_FILENO);
                close(pipefd[1]);
            }
            apply_redirects(pipe.stages[i]);

            if (is_builtin(pipe.stages[i].argv[0])) {
                int rc = run_builtin(pipe.stages[i]);
                _exit(rc);
            }
            auto argv = to_c_argv(pipe.stages[i].argv);
            execvp(argv[0], argv.data());
            std::cerr << "msh: " << argv[0] << ": " << std::strerror(errno) << '\n';
            _exit(127);
        }

        pids.push_back(pid);

        if (prev_read != -1) close(prev_read);
        if (!last) {
            close(pipefd[1]);
            prev_read = pipefd[0];
        }
    }

    if (pipe.background) {
        std::cout << "[bg " << pids.back() << "]\n";
        return 0;
    }

    int status = 0;
    for (pid_t p : pids) waitpid(p, &status, 0);
    if (WIFEXITED(status))   return WEXITSTATUS(status);
    if (WIFSIGNALED(status)) return 128 + WTERMSIG(status);
    return 1;
}

}
```

## Wire it up

In `shell.cpp`:

```cpp
int Shell::execute(const std::string& line) {
    try {
        auto tokens = tokenize(line);
        if (tokens.empty()) return 0;
        Pipeline pipe = parse(std::move(tokens));
        return msh::execute(pipe);
    } catch (const std::exception& e) {
        std::cerr << "msh: " << e.what() << '\n';
        return 1;
    }
}
```

Add `expand.cpp` to your CMake target list.
